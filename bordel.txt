  if [ $sc_out -ge 1 ];then
    see_it=$((see_it+1))
  fi
  
  if [ see_it -ge 5 ]; then
    uci set wireless.@wifi-iface[0].disabled=1; wifi; ## here instead of zero comes the exact value
  fi
  
  sleep 2
done
